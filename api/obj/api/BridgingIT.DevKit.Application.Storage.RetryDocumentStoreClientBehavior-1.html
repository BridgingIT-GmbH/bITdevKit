Examples

services.AddEntityFrameworkDocumentStoreClient<Person, AppDbContext>()
    .WithBehavior((inner, sp) => new RetryDocumentStoreClientBehavior<Person>(
        sp.GetRequiredService<ILoggerFactory>(),
        inner,
        new RetryDocumentStoreClientBehaviorOptions
        {
            Attempts = 5,
            Backoff = TimeSpan.FromMilliseconds(100),
            BackoffExponential = true
        }));
