Examples

var document = new ComponentDiagramBuilder(DiagramDirection.LeftToRight)
    .AddComponent("Api", "Orders API")
    .AddComponent("Database", kind: DiagramNodeKind.Database)
    .AddDependency("Api", "Database", "reads")
    .Build();
