public static DateTimeOffset ToDateTimeOffset(this DateTime localDateTime, TimeZoneInfo timeZone, InvalidTimePolicy invalidTimePolicy = InvalidTimePolicy.Throw, AmbiguousTimePolicy ambiguousTimePolicy = AmbiguousTimePolicy.Throw)

Parameters

localDateTime DateTime

The local wall-clock value.

timeZone TimeZoneInfo

The time zone whose rules are used.

invalidTimePolicy InvalidTimePolicy

ambiguousTimePolicy AmbiguousTimePolicy

DateTimeOffset
