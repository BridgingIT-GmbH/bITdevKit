Class Either<T1, T2>

Namespace
BridgingIT.DevKit.Common

    var result = await either.MatchAsync(

    // Handle not found
}

// With cancellation

    async num => await ProcessNumberAsync(num, cts.Token),
    async str => await ProcessStringAsync(str, cts.Token)
);
